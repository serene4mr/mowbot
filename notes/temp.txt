To run simulation demo:

- run gzsim from other container
- run ros2 launch mowbot_launch gzsim_bringup.launch.py
- run ros2 launch mowbot_launch mowbot.launch.py use_sim_time:=true launch_robot:=false


source ./install/setup.bash

ros2 launch mowbot_launch gzsim_bringup.launch.py
ros2 launch mowbot_launch mowbot.launch.py use_sim_time:=true launch_robot:=false
ros2 launch mowbot_launch mapviz_localization.launch.py
ros2 run mowbot_mission_bridge mission_bridge_node --ros-args --params-file src/core/mowbot_sdk/misc/mowbot_mission_bridge/config/mission_params.yaml

ros2 launch nav2_bringup rviz_launch.py 

ros2 launch mowbot_launch mowbot.launch.py \
  robot_id:=mowbot_001 \
  data_path:=/workspaces/mowbot/src/mowbot_data \
  launch_robot:=false \
  launch_sensing:=true \
  launch_sensing_driver:=false \
  launch_localization:=true \
  launch_nav2:=true \
  launch_app:=true \
  use_sim_time:=true


# Terminal 1 - Robot 1 (Domain 0)
export ROS_DOMAIN_ID=0
ros2 launch mowbot_launch gzsim_bringup.launch.py \
  robot_id:=mowbot_001 \
  manufacturer:=MowbotTech \
  data_path:=/workspaces/mowbot/src/mowbot_data \
  x:=-15.85 y:=5.64 yaw:=0.9 

# Terminal 2 - Robot 1 Stack (Domain 0)
export ROS_DOMAIN_ID=0
ros2 launch mowbot_launch mowbot.launch.py \
  robot_id:=mowbot_001 \
  manufacturer:=MowbotTech \
  data_path:=/workspaces/mowbot/src/mowbot_data \
  launch_robot:=false launch_sensing:=true launch_sensing_driver:=false \
